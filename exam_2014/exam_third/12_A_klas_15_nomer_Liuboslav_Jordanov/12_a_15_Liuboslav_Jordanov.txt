CREATE TABLE IF NOT EXISTS Category(id int AUTO_INCREMENT, date_created_on date, name varchar(50), PRIMARY KEY (id))
CREATE TABLE IF NOT EXISTS User(id int AUTO_INCREMENT, password varchar(50), description varchar(255), income float, PRIMARY KEY (id))
CREATE TABLE IF NOT EXISTS Tag(id int AUTO_INCREMENT, second_priority float, name varchar(50), PRIMARY KEY (id))
